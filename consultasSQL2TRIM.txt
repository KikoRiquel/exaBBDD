/* 1 COMPLETO */
SELECT NOMBRE_DISCO, GRUPO, DISCOGRAFICA, FECHA_PUBLICACION
FROM DISCOS
WHERE TO_CHAR(FECHA_PUBLICACION) LIKE '%83'
ORDER BY FECHA_PUBLICACION ASC;

/* 2  INCOMPLETO NO MUESTRA EL TITULO SOLO VENTAS */
SELECT G.NOMBRE_GRUPO, MAX(D.MILLONES_VENDIDOS)
FROM GRUPOS G
JOIN DISCOS D ON D.GRUPO = G.COD_GRUPO
GROUP BY G.NOMBRE_GRUPO
ORDER BY MAX(D.MILLONES_VENDIDOS) DESC;

/* 3 INCOMPLETO SOLO MUESTRA MAYOR NUMERO DE DISCOS Y NO EL NOMBRE */
SELECT MAX(COUNT(*))
FROM DISCOS
GROUP BY GRUPO;

/* 4 COMPLETO */
SELECT G.NOMBRE_GRUPO AS GRUPO, MAX( D.FECHA_PUBLICACION) AS ULTIMA_PUBLICACION
FROM GRUPOS G
JOIN DISCOS D ON D.GRUPO = G.COD_GRUPO
WHERE G.COD_GRUPO IN
( SELECT GRUPO
  FROM MUSICOS
  WHERE NOMBRE_MUSICO LIKE '%Ian%')
GROUP BY G.NOMBRE_GRUPO;

/* 5 INCOMPLETO NO MUESTRA NOMBRE Y FECHA SOLO CANTIDAD */
SELECT G.NOMBRE_GRUPO, MAX(D.MILLONES_VENDIDOS)
FROM DISCOS D
JOIN GRUPOS G
ON D.GRUPO=G.COD_GRUPO
GROUP BY G.NOMBRE_GRUPO;

/* 6 COMPLETO */
SELECT G.NOMBRE_GRUPO AS GRUPO, COUNT(D.COD_DISCO) AS DISCOS
FROM GRUPOS G
JOIN DISCOS D
ON G.COD_GRUPO = D.GRUPO
GROUP BY G.NOMBRE_GRUPO;

/* 7 COMPLETO */
SELECT D.NOMBRE_DISCO, G.NOMBRE_GRUPO
FROM DISCOS D
JOIN GRUPOS G
ON D.GRUPO=G.COD_GRUPO
WHERE D.FECHA_PUBLICACION > TO_DATE('01/01/1975','DD/MM/YY')
AND D.FECHA_PUBLICACION < TO_DATE('31/12/1988','DD/MM/YY');
